
\section{Introdução}
\subsection{Sobre o que se trata?}
\frame{
    \frametitle{Tópicos}
    \begin{itemize}
    \item Network coding, o estudo das técnicas de agrupamento de
    pacotes para transmissão em redes.
    \item Redes sociais; em particular, redes sociais móveis.
    \end{itemize}
}

\subsection{Redes sociais móveis}
\frame{
    \frametitle{Redes sociais móveis}
    \begin{itemize}
    \item Rede sujeita a atrasos e desconexões.
    \item DTN (\textit{Delay Tolerant Network})
    \item Roteamento só ocorre quando usuários estão fisicamente
    próximos.
    \end{itemize}
}
