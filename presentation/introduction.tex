
\section{Introdução}
%\subsection{Sobre o que se trata?}
\frame{
    \frametitle{Tópicos}
    \begin{itemize}
    \item Redes sociais; em particular, redes sociais móveis.
    \item Codificação em rede, o estudo das técnicas de agrupamento de
    pacotes para transmissão em redes.
    \end{itemize}
}

%\subsection{Redes sociais móveis}
\frame{
    \frametitle{Redes sociais móveis}
    \begin{itemize}
    \item Rede sujeita a atrasos e desconexões.
    \item DTN (\textit{Delay Tolerant Network})
    \item Roteamento só ocorre quando usuários estão fisicamente
    próximos.
    \end{itemize}
}

%\subsection{A ideia}
\frame{
    \frametitle{A ideia}
    \begin{itemize}
    \item Criar um protocolo de roteamento de mensagens baseado
    em informações de relacionamentos entre pessoas.
    \item Diminuir o número de mensagens enviadas.
    \end{itemize}
}
